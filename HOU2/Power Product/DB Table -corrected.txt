CREATE TABLE PrimaryPowerProduct(
cID VARCHAR(6),
sID VARCHAR(6),
panelName VARCHAR(8),
power_type VARCHAR(11),
mau SMALLINT(5),
location VARCHAR(5),
row VARCHAR(2),
cab VARCHAR(2),
INDEX(mau),
INDEX(row),
INDEX(cab),
PRIMARY KEY(sID))ENGINE MyISAM;
